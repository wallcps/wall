    $('.update-profile').click(function(){
        var about = $('#about').val();
        var contact = $('#contact').val();
        var email = $('#email').val();
        var skills = $('#skills').val();
        var interest = $('#interest').val();
        $.ajax({
            type:'POST',
            url:'<?php echo $base_url; ?>project/ajax-project.php',
            data:{
               about    :about,
               contact  :contact,
               email    :email,
               skills    :skills,
               interest    :interest,
               uid      : "<?php echo $uid; ?>",
               post_type:'edit_profile',
            },
            success:function(data){
                location.reload();
            },error:function(data){
                alert(data);
            }
        });
    });

</script>
